================================================================================
                              H1 Pipeline
================================================================================

【Teacher Policy】
训练：
../IsaacLab/isaaclab.sh -p scripts/rsl_rl/train_teacher_policy.py \
    --robot h1 \
    --num_envs 2048 \
    --teacher_policy.max_iterations 20000 \
    --reference_motion_path neural_wbc/data/data/motions/multitask_h1.pkl \
    --headless

# TensorBoard 日志查看
../IsaacLab/isaaclab.sh -p -m tensorboard.main \
  --logdir=logs/teacher

断点重训：
../IsaacLab/isaaclab.sh -p scripts/rsl_rl/train_teacher_policy.py \
    --robot h1 \
    --num_envs 2048 \
    --teacher_policy.max_iterations 10000 \
    --reference_motion_path neural_wbc/data/data/motions/multitask_h1.pkl \
    --resume \
    --headless

播放：
../isaaclab/isaaclab.sh -p scripts/rsl_rl/play_teacher_policy.py \
    --robot h1 \
    --num_envs 16 \
    --reference_motion_path neural_wbc/data/data/motions/multitask_h1.pkl \
    --teacher_policy.resume_path neural_wbc/data/data/policy/h1:teacher \
    --teacher_policy.checkpoint model_10000.pt
